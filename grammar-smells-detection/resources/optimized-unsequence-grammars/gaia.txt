OrganizationalStructure ::= Organization+ ;
Organization ::= Organization Organization AgentType+ OrganizationalRule+ ;
AgentType ::= Service+ AgentType AgentType Role+ ;
Service ::= String String String String ;
Role ::= Activity+ Communication+ Responsibility+ AssociationConnector ;
AssociationConnector ::= Resource+ Role Permission+ ;
Permission ::= AssociationConnector ;
Activity ::= ε ;
Responsibility ::= LivenessProperty ;
Responsibility ::= SafetyProperty ;
Responsibility ::= LivenessProperty | SafetyProperty ;
LivenessProperty ::= ε ;
Resource ::= String String Action+ Environnement ;
Environnement ::= Resource+ ;
Action ::= String ;
Communication ::= OrganizationalRule+ Protocol ;
Protocol ::= String String String String String String ;
String ::= STR ;
OrganizationalRule ::= SafetyRule ;
OrganizationalRule ::= LivenessRule ;
OrganizationalRule ::= SafetyRule | LivenessRule ;
SafetyRule ::= ε ;
SafetyProperty ::= ε ;
LivenessRule ::= ε ;
